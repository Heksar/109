# Write a function to compute the area of a circle
# -------------------------------------------------------------
def areaOfCircle(radius) :
    # pi = 3.1415
    return pi * radius * radius

area = areaOfCircle(1)
print('radius = 1 and area = ', area)

# 1. Can you use return in a loop?  Try it with perfectcuberoot function.
# 2. A function call is an expression
# 3. A function call has a value -- what is the value of areaOfCircle(3)?

def perfectcuberoot(n) :
    ans = None
    for guess in range(n + 1) :
        if guess ** 3 == n :
            return guess
        if guess ** 3 > n :
            break
    return ans

for i in range(9) :
    print('i = ', i, 'cube root = ', perfectcuberoot(i))

# Chapter 4: Functions: binding of formal parameters, example hello(name)
# ------------------------------------------------------------
# Write a function to return string "Hello X", where X is the given name
# -------------------------------------------------------------
def hello(first, last, reverse = False) :
    if reverse :
        print('Hello ', last, first)
    else :
        print('Hello ', first, last)

hello('Sadaf', 'Mustafiz', False)
hello('Sadaf', 'Mustafiz', True)
