# --------------------------------------------------------------
def printmx(mx) :
    for i in range(len(mx)) :
        printrow(mx[i])
    print('-------------------')
def printrow(row) :
    for i in range(len(row)) :
        print('{:4d}'.format(row[i]), end = '')
    print()
